+def _minmax(values: np.ndarray) -> np.ndarray:
+    if values.size == 0:
+        return values
+    min_v = values.min()
+    max_v = values.max()
+    span = max_v - min_v
+    if span <= 1e-12:
+        return np.zeros_like(values, dtype=np.float32)
+    return (values - min_v) / span
